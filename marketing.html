from flask import Flask, request, jsonify
from flask_cors import CORS
import torch
from diffusers import AutoencoderKLWan, WanPipeline
from diffusers.schedulers.scheduling_unipc_multistep import UniPCMultistepScheduler

app = Flask(__name__)
CORS(app)

pipe = WanPipeline.from_pretrained(
    "Wan-AI/Wan2.1-T2V-1.3B", torch_dtype=torch.float16
).to("cuda")

@app.route("/generate", methods=["POST"])
def generate():
    data = request.json
    output = pipe(
        prompt=data.get("prompt",""),
        negative_prompt=data.get("negative_prompt",""),
        num_frames=data.get("num_frames", 48),
        guidance_scale=data.get("guidance_scale", 6)
    ).frames[0]
    # Save and return base64
    import tempfile, base64
    from diffusers.utils import export_to_video
    tmp = tempfile.NamedTemporaryFile(suffix=".mp4", delete=False)
    export_to_video(output, tmp.name, fps=16)
    return jsonify({"video": base64.b64encode(open(tmp.name,"rb").read()).decode()})

app.run(host="0.0.0.0", port=7860)

from flask import Flask, request, jsonify
from flask_cors import CORS
import torch, io, base64
from diffusers import FluxPipeline

app = Flask(__name__)
CORS(app)

pipe = FluxPipeline.from_pretrained(
    "black-forest-labs/FLUX.1-dev",
    torch_dtype=torch.bfloat16
).to("cuda")

@app.route("/generate", methods=["POST"])
def generate():
    data = request.json
    w, h = map(int, data.get("resolution","1024x1024").split("x"))
    images = pipe(
        prompt=data["prompt"],
        negative_prompt=data.get("negative_prompt",""),
        height=h, width=w,
        guidance_scale=float(data.get("guidance_scale", 3.5)),
        num_inference_steps=int(data.get("num_steps", 28)),
        num_images_per_prompt=int(data.get("count",1)),
    ).images
    imgs_b64 = []
    for img in images:
        buf = io.BytesIO()
        img.save(buf, format="PNG")
        imgs_b64.append(base64.b64encode(buf.getvalue()).decode())
    return jsonify({"images": imgs_b64})

app.run(host="0.0.0.0", port=7861)
Ejecutar: python flux_server.py
